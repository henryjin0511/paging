Paging 原生javascript分页插件

加载项


<script type="text/javascript" src="/dist/paging.js"></script>
<link rel="stylesheet" type="text/css" href="/dist/paging.css" />
* 实际运用中建议css自定义

默认


var paging1 = paging({
	cont:'pagination1',//id名或者直接传入原生dom对象
	pages:12,//总页数
	groups:5,//居中显示数量
	curr:6//当前页数
});


默认 增加总页数显示


var paging1 = paging({
	cont:'pagination1',//id名或者直接传入原生dom对象
	pages:12,
	groups:5,
	curr:6,
	total:true
});


默认 增加跳页功能


var paging1 = paging({
	cont:'pagination1',//id名或者直接传入原生dom对象
	pages:12,
	groups:5,
	curr:6,
	total:true,
	skip:true
});


只含上下页，当前页


var paging1 = paging({
	cont:'pagination1',//id名或者直接传入原生dom对象
	pages:12,
	groups:1,
	curr:6
});


上下页


var paging1 = paging({
	cont:'pagination1',//id名或者直接传入原生dom对象
	pages:12,
	groups:false,
	curr:6
});


回调及方法


var paging1 = paging({
	cont:'pagination1',//id名或者直接传入原生dom对象
	pages:12,
	groups:false,

});
